```tsx
useEffect(() => {
  async function enumerate() {
    try {
      // Request permission first so device labels are populated
      const stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: false })
      stream.getTracks().forEach(t => t.stop()) // Release immediately after permission

      const devices = await navigator.mediaDevices.enumerateDevices()
      const videoDevices = devices.filter(d => d.kind === 'videoinput')
      setCameras(videoDevices)
      if (videoDevices.length > 0) setSelectedCamera(videoDevices[0].deviceId)
    } catch (e) {
      console.error('Camera enumerate failed', e)
    }
  }
  enumerate()
}, [])
```

MediaRecorder setup (called right before countdown reaches 3):
```tsx
const canvasStream = canvasRef.current!.captureStream(30)
const mimeType = MediaRecorder.isTypeSupported('video/webm;codecs=vp9')
  ? 'video/webm;codecs=vp9'
  : MediaRecorder.isTypeSupported('video/webm')
  ? 'video/webm'
  : 'video/mp4'

const recorder = new MediaRecorder(canvasStream, { mimeType })
const chunks: Blob[] = []
recorder.ondataavailable = (e) => { if (e.data.size > 0) chunks.push(e.data) }
recorder.onstop = async () => {
  const rawBlob = new Blob(chunks, { type: mimeType })
  // Fix duration using webm-fix-duration
  const { fixWebmDuration } = await import('webm-fix-duration')
  const durationMs = timerMsRef.current
  const fixedBlob = await fixWebmDuration(rawBlob, durationMs)
  setRecordedBlob(fixedBlob) // RecordContext setter
  setMimeType(mimeType)      // RecordContext setter
  router.push('/record/playback')
}
mediaRecorderRef.current = recorder
recorder.start()
```

SCREEN WAKE LOCK:
```tsx
const wakeLockRef = useRef<WakeLockSentinel | null>(null)

// Acquire when recording starts
async function acquireWakeLock() {
  try {
    wakeLockRef.current = await navigator.wakeLock.request('screen')
  } catch (e) {
    console.warn('Wake Lock not available', e)
  }
}

// Release when recording stops
function releaseWakeLock() {
  wakeLockRef.current?.release()
  wakeLockRef.current = null
}
```

Call acquireWakeLock() when MediaRecorder.start() is called.
Call releaseWakeLock() in the stop handler.

Also add a visibilitychange listener to re-acquire the lock if the page regains visibility while recording:
```tsx
useEffect(() => {
  const handler = async () => {
    if (document.visibilityState === 'visible' && isRecording && !wakeLockRef.current) {
      await acquireWakeLock()
    }
  }
  document.addEventListener('visibilitychange', handler)
  return () => document.removeEventListener('visibilitychange', handler)
}, [isRecording])
```

AUDICONTEXT BEEPS (minute marker):
```tsx
const audioCtxRef = useRef<AudioContext | null>(null)

function playBeep() {
  if (!audioCtxRef.current) {
    audioCtxRef.current = new AudioContext()
  }
  const ctx = audioCtxRef.current
  if (ctx.state === 'suspended') ctx.resume()

  const oscillator = ctx.createOscillator()
  const gainNode = ctx.createGain()
  oscillator.connect(gainNode)
  gainNode.connect(ctx.destination)

  oscillator.frequency.value = 880 // A5
  oscillator.type = 'sine'
  gainNode.gain.setValueAtTime(0.3, ctx.currentTime)
  gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.3)

  oscillator.start(ctx.currentTime)
  oscillator.stop(ctx.currentTime + 0.3)
}
```

Trigger beep logic: In the rAF loop (or a useEffect watching timerMs), check if beepEveryMinute is true AND the timer just crossed a whole minute boundary:
```tsx
// In rAF loop, track lastBeepMinute in a ref
const lastBeepMinuteRef = useRef(-1)

if (beepEveryMinute && currentMinute > 0 && currentMinute !== lastBeepMinuteRef.current) {

}
```
