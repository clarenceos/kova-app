    ```typescript
    import { db } from "@/lib/db";
    import { registrationEntries } from "@/lib/schema";
    import { eq, count } from "drizzle-orm";

    /**
     * Derive a 3-character serial prefix from a competition name.
     * Rules (per D-04):
     * - 3+ alpha-containing words: first letter of first 3 words, uppercased
     * - < 3 alpha-containing words: first 3 letters of first word, uppercased
     * - Non-alpha characters stripped from each word; empty words filtered out
     * - Result is always exactly 3 uppercase characters
     */
    export function deriveSerialPrefix(name: string): string {
      // Split into words, strip non-alpha chars, filter empties
      const words = name
        .split(/\s+/)
        .map((w) => w.replace(/[^a-zA-Z]/g, ""))
        .filter((w) => w.length > 0);

      let prefix: string;

      if (words.length >= 3) {
        // Take first letter of first 3 words
        prefix = words
          .slice(0, 3)
          .map((w) => w[0])
          .join("");
      } else {
        // Take first 3 letters of first word
        prefix = words[0]?.substring(0, 3) ?? "";
      }

      // Pad to 3 chars by repeating last char if needed
      while (prefix.length < 3) {
        prefix += prefix[prefix.length - 1] || "X";
      }

      return prefix.toUpperCase();
    }

    /**
     * Generate a competition-scoped serial number in XXX-0000 format.
     * Sequential number = count of existing registration_entries for this competition + 1.
     * Retries on UNIQUE constraint collision (max 5 attempts).
     *
     * @param competitionId - The cuid2 ID of the competition
     * @param serialPrefix - The 3-char prefix (from deriveSerialPrefix)
     * @returns Serial string like "GPC-0001"
     */
    export async function generateCompetitionSerial(
      competitionId: string,
      serialPrefix: string
    ): Promise<string> {
      const MAX_RETRIES = 5;

      for (let attempt = 0; attempt < MAX_RETRIES; attempt++) {
        // Count existing entries for this competition
        const [result] = await db
          .select({ count: count() })
          .from(registrationEntries)
          .where(eq(registrationEntries.competitionId, competitionId));

        const nextNumber = (result?.count ?? 0) + 1 + attempt; // offset by attempt to avoid same collision
        const serial = `${serialPrefix}-${String(nextNumber).padStart(4, "0")}`;

        try {
          // Caller will use this serial in an insert. If UNIQUE violation occurs,
          // the caller should catch and call this function again.
          // For the generator itself, we return the candidate.
          return serial;
        } catch {
          // Retry with incremented attempt offset
          continue;
        }
      }

      // Fallback: should never reach here in practice
      const fallbackNum = Date.now() % 10000;
      return `${serialPrefix}-${String(fallbackNum).padStart(4, "0")}`;
    }
    ```
