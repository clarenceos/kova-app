```typescript
export interface SchedulerEntry {
  registrantId: string;
  firstName: string;
  lastName: string;
  gender: 'Male' | 'Female';
  bodyWeightKg: number;
  country: string;
  club: string | null;
  coach: string | null;
  entryId: string;
  event: 'LC' | 'JERK' | 'SNATCH';
  bellWeight: string;
  duration: number;
  serial: string;
}

export interface SchedulerInput {
  entries: SchedulerEntry[];
  numPlatforms: number;
  startTimeMinutes: number;
  blockDuration?: number;      // default 10
  transitionDuration?: number;  // default 5
  minRestBlocks?: number;       // default 2
}

export interface PlatformSlot {
  entryId: string;
  registrantId: string;
  firstName: string;
  lastName: string;
  event: 'LC' | 'JERK' | 'SNATCH';
  bellWeight: string;
  duration: number;
  gender: 'Male' | 'Female';
  bodyWeightKg: number;
  weightClass: string;
  club: string | null;
  coach: string | null;
  country: string;
  serial: string;
}

export interface TimeBlock {
  blockNumber: number;
  startTime: number;
  endTime: number;
  platforms: (PlatformSlot | null)[];
}

export interface RestConflict {
  type: 'REST';
  athleteName: string;
  blockNumbers: [number, number];
  gap: number;
  minRequired: number;
  event: string;
  bellWeight: string;
}

export interface CoachConflict {
  type: 'COACH';
  coachName: string;
  studentName: string;
  blockNumber: number;
  event: string;
  bellWeight: string;
}

export type Conflict = RestConflict | CoachConflict;

export interface ScheduleResult {
  timeBlocks: TimeBlock[];
  conflicts: Conflict[];
  estimatedFinishTime: number;
}
```

    Create `lib/queue/scheduler.ts` exporting a single `schedule` function:

    ```typescript
    import type { SchedulerInput, ScheduleResult, SchedulerEntry, TimeBlock, PlatformSlot, Conflict, RestConflict, CoachConflict } from './types';
    import { getWeightClass } from './weightClass';

    export function schedule(input: SchedulerInput): ScheduleResult { ... }
    ```

    The function has three internal phases:

    **Phase A: Sort entries** (per D-07, D-08, SCHED-02)

    Define event priority map: `{ LC: 0, JERK: 1, SNATCH: 2 }`.

    **IMPORTANT — Weight class sort requires a numeric-aware comparator, NOT plain alphabetical string sort.**

    Plain `localeCompare` or string comparison of weight class labels is BROKEN for super-heavyweight brackets: `'80+kg'.localeCompare('80kg')` returns `-1`, meaning `80+kg` would incorrectly sort before `80kg`. Similarly `'95+kg'` before `'95kg'`.

    Implement a `weightClassSortKey(wc: string): number` helper function inside scheduler.ts (not exported — internal only):
    ```typescript
    /** Extracts a numeric sort key from a weight class label.
     *  '52kg' -> 52, '80kg' -> 80, '80+kg' -> 80.5, '95+kg' -> 95.5
     *  The +0.5 ensures super-heavyweight sorts after its bounded counterpart. */
    function weightClassSortKey(wc: string): number {
      const num = parseFloat(wc);  // parseFloat('80+kg') -> 80, parseFloat('80kg') -> 80
      return wc.includes('+') ? num + 0.5 : num;
    }
    ```

    Sort comparator chain (each step breaks ties for the next):
    1. Event priority: LC(0) < JERK(1) < SNATCH(2) — ascending
    2. Duration: 10 before 5 — descending (b.duration - a.duration)
    3. Gender: Female before Male — Female < Male alphabetically, so `a.gender.localeCompare(b.gender)` works since 'Female' < 'Male'
    4. Weight class: derive via `getWeightClass(entry.gender, entry.bodyWeightKg)`, then compare using `weightClassSortKey(wcA) - weightClassSortKey(wcB)`. This correctly places 80+kg after 80kg and 95+kg after 95kg.
    5. Body weight (D-08 tiebreaker): `a.bodyWeightKg - b.bodyWeightKg` (lighter first)

    Create a sorted copy (do NOT mutate the input array).

    **Phase B: Assign to blocks** (SCHED-03)

    Destructure with defaults: `blockDuration = 10`, `transitionDuration = 5`, `minRestBlocks = 2`.

    Iterate sorted entries. Fill `numPlatforms` slots per block (greedy). When a block is full, start a new block.

    Each TimeBlock:
    - `blockNumber`: 1-indexed
    - `startTime`: `startTimeMinutes + (blockIndex * (blockDuration + transitionDuration))`
    - `endTime`: `startTime + blockDuration`
    - `platforms`: array of `PlatformSlot | null`, length `numPlatforms`

    Each PlatformSlot is built from the SchedulerEntry plus `weightClass: getWeightClass(entry.gender, entry.bodyWeightKg)`.

    Pad the last block's platforms array with `null` if fewer entries than platforms.

    `estimatedFinishTime`: if no blocks, return `startTimeMinutes`. Otherwise, last block's `endTime`.

    **Phase C: Detect conflicts** (SCHED-04, SCHED-05)

    **REST conflicts (D-09, SCHED-04):**
    Build a map: `registrantId -> blockNumber[]` (a registrant can appear in multiple blocks if they have multiple entries, e.g., biathlon).
    For each registrant with 2+ block numbers, check all pairs. If `blockNumbers[j] - blockNumbers[i] < minRestBlocks` (strict less-than per D-09), emit a RestConflict:
    ```typescript
    {
      type: 'REST',
      athleteName: `${firstName} ${lastName}`,
      blockNumbers: [blockNumbers[i], blockNumbers[j]],
      gap: blockNumbers[j] - blockNumbers[i],
      minRequired: minRestBlocks,
      event: entry.event,     // event of the later entry
      bellWeight: entry.bellWeight,
    }
    ```

    **COACH conflicts (D-01, D-02, D-03, SCHED-05):**
    For each block, collect all entries in that block. For each entry that has a non-null `coach` field:
    - Normalize coach name: `coach.toLowerCase().trim()`
    - For every OTHER entry in the same block, compute full name: `(firstName + ' ' + lastName).toLowerCase().trim()`
    - If normalized coach matches normalized full name, emit a CoachConflict:
    ```typescript
    {
      type: 'COACH',
      coachName: coach,           // original case from entry
      studentName: `${studentEntry.firstName} ${studentEntry.lastName}`,
      blockNumber: block.blockNumber,
      event: studentEntry.event,
      bellWeight: studentEntry.bellWeight,
    }
    ```

    Wait — re-read D-03: "Coach field is `registrant.coach`. Match against `firstName + " " + lastName` of other registrants in the same block." The student is the one WITH the coach field. The coach (as an athlete) is the one whose name matches. So:
    - The entry with `coach` field set = the student
    - The entry whose `firstName + " " + lastName` matches the coach value = the coach (as an athlete)
    - CoachConflict.coachName = the matched athlete's full name
    - CoachConflict.studentName = the entry with the coach field set

    Actually, re-reading D-05 more carefully: `coachName` and `studentName` are both present. The student has the `coach` field. The coach is the athlete whose name matches. So:
    - `coachName`: the value from `entry.coach` (the coach's name as entered by the student)
    - `studentName`: `${entry.firstName} ${entry.lastName}` (the student who listed the coach)
